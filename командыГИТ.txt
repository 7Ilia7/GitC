Команды в Git Bash консолиGitHub
git clone https -- склонировать репозиторий.
git status -- проверка состояния файлов.
git add Имя файла -- показать файл гиту 
       --git add . -- показать все файлы
1.git add . показать все файлы гиту 
2.git commit -m "new" - зафиксировать изменения 
       --git commit -a
3.git push -- запушить изменения на удаленный репозиторий
	--git push --all -- запушить все и вручную добавить ветки на гитхаб
clear — Очистить консоль
$SHELL- что бы консоль в конце не закрывалась

Если не коммитит то прописать 
git config --global user.name "Имя репозиториякуда комитим"
git config --global user.email "@на котором аккаунт гита"

Навигация
pwd — Показать текущий каталог
ls - Показать файлы в данной папке, кроме скрытых
ls -f — Показать файлы в данной папке, включая и скрытые
cd c:/ — Перейти в конкретный каталог
cd - — Вернуться назад
cd .. — Выйти на 1 уровень вверх
cd ../.. — Выйти на 2 уровня вверх
vim - зайти в файл(для изменения)	
q! - выйти из файла без сохранения
grep 'па' main1.js -  поиск по той и файле дерриктории в которой ты находишься

Создание каталогов
mkdir — Создать папку
mkdir dur1 | touch dur1/n.log создать папку и в ней файл
cd !$ — Перейти в только что созданную папку
mkdir -p {app1,app2} — Создать сразу несколько папок
mkdir -p app/{css,js} — Создать сразу несколько вложенных папок

Создание файлов
touch index.html — Создать файл index.html
touch css/main.css js/main.js index.html — Создать сразу несколько файлов, никаких лишних пробелов быть не должно
cp test.txt ../2/test.txt скопировать файл с переходом на уровень выше
cp - копирование файла
mv qa1.txt qa2.txt переиминование файла 
mv - перемещение файла в системе (mv f_1/j_test1.json fff_1/j_test1.json)
cat > tt2.txt создание и запись файла сразу
cat ttn1.txt >> ttn3.txt дописать 
cat ttn1.txt > ttn3.txt перезапись

Удаление файлов
touch — позволяет создавать файлы
rm test — Удалить пустую папку test
rm -r test — Удалить папку test с файлами внутри неё
rmdir - Удалить пустую папку

Перемещение файлов
find qa4 main.js найти файт в этой папке с атрибутом имя
mv app1/txt.txt app2/txt.txt — перемесqaтить конкретній файл.
mv app1/*.* app2 — Переместить все файлы из папки app1 в папку app2 
cp qa3/txt.txt qa1/txt.txt скопировать файл в другую директорию.
